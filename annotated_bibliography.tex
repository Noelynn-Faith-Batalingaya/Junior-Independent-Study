% This is a template you can use for your annotated bibliography. The only
% things you should have to edit in this file are the title and author. You'll
% write the annotations themselves in the "annote" fields of the entries in
% bibliography.bib.
%
% When using a .bib file, compiling a document takes more steps. You need to
% compile it as you normally would, then run bibtex, then compile it as you
% normally would twice again. The LaTeX system does a little more work on each
% pass until all the references are properly created.
%
% So if you're compiling documents on the command line, you'll run the
% following:
%
% pdflatex annotated_bibliography
% bibtex annotated_bibliography
% pdflatex annotated_bibliography
% pdflatex annotated_bibliography
%
% Alternately, you may compile using latexmk on the command line. Then,
% you can run the following:
% 
% latexmk -pdf annotated_bibliography.tex
%
%
% If you are using a graphical LaTeX environment, you should be able to do the
% same sequence of commands in whatever manner the environment provides. If you
% screw up something in the .bib file, you might need to delete the .aux, .bbl,
% and .blg files that are generated during the compilation process. If you're
% using a graphical environment there should be a way to clean up the build
% files.

\documentclass[12pt]{article}
\usepackage{setspace}
\singlespace
\usepackage[left=1in,right=1in,top=1in,bottom=1in]{geometry}

\title{\textbf{Annotated Bibliography Template}}
\author{Your Name}

\begin{document}

\maketitle

% Using this makes it so that all the references show up regardless of whether
% or not they were actually cited in the rest of the document. Since we are
% only displaying the bibliography we won't be citing anything anywhere, so we
% need this.
\nocite{*}

\bibliographystyle{plain-annote}
\bibliography{bibliography.bib}

\end{document}
